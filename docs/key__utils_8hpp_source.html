76 // to_key defines a conversion from a type to a sequence of bytes whose lexicograpical
77 // ordering is the same as the ordering of the original type.
78 //
79 // For any two objects of type T, a and b:
80 //
81 // - key(a) < key(b) iff a < b
82 // - key(a) is not a prefix of key(b)
83 //
84 // Overloads of to_key for user-defined types can be found by Koenig (ADL) lookup.
85 //
86 // to_key is specialized for the following types
87 // - std::string and std::string_view
88 // - std::vector, std::list, std::deque
89 // - std::tuple
90 // - std::array
91 // - std::optional
92 // - std::variant
93 // - Arithmetic types
94 // - Scoped enumeration types
95 // - Reflected structs
96 // - All smart-contract related types defined by abieos

179 // The first byte holds:
180 // 0-4 1's (number of additional bytes) 0 (terminator) bits
181 //
182 // The number is represented as big-endian using the low order
183 // bits of the first byte and all of the remaining bytes.
184 //
185 // Notes:
186 // - values must be encoded using the minimum number of bytes,
187 // as non-canonical representations will break the sort order.
188 template <typename S>
189 void to_key_varuint32(std::uint32_t obj, datastream<S>& stream) {
190  int num_bytes;
191  if (obj < 0x80u) {
192  num_bytes = 1;
193  } else if (obj < 0x4000u) {
194  num_bytes = 2;
195  } else if (obj < 0x200000u) {
196  num_bytes = 3;
197  } else if (obj < 0x10000000u) {
198  num_bytes = 4;
199  } else {
200  num_bytes = 5;
201  }
202 
203  stream.write(
204  static_cast<char>(~(0xFFu >> (num_bytes - 1)) | (num_bytes == 5 ? 0 : (obj >> ((num_bytes - 1) * 8)))));
205  for (int i = num_bytes - 2; i >= 0; --i) { stream.write(static_cast<char>((obj >> i * 8) & 0xFFu)); }
206 }

208 // for non-negative values
209 // The first byte holds:
210 // 1 (signbit) 0-4 1's (number of additional bytes) 0 (terminator) bits
211 // The value is represented as big endian
212 // for negative values
213 // The first byte holds:
214 // 0 (signbit) 0-4 0's (number of additional bytes) 1 (terminator) bits
215 // The value is adjusted to be positive based on the range that can
216 // be represented with this number of bytes and then encoded as big endian.
217 //
218 // Notes:
219 // - negative values must sort before positive values
220 // - For negative value, numbers that need more bytes are smaller, hence
221 // the encoding of the width must be opposite the encoding used for
222 // non-negative values.
223 // - A 5-byte varint can represent values in $[-2^34, 2^34)$. In this case,
224 // the argument will be sign-extended.
225 template <typename S>
226 void to_key_varint32(std::int32_t obj, datastream<S>& stream) {
227  static_assert(std::is_same_v<S, void>, "to_key for varint32 has been temporarily disabled");
228  int num_bytes;
229  bool sign = (obj < 0);
230  if (obj < 0x40 && obj >= -0x40) {
231  num_bytes = 1;
232  } else if (obj < 0x2000 && obj >= -0x2000) {
233  num_bytes = 2;
234  } else if (obj < 0x100000 && obj >= -0x100000) {
235  num_bytes = 3;
236  } else if (obj < 0x08000000 && obj >= -0x08000000) {
237  num_bytes = 4;
238  } else {
239  num_bytes = 5;
240  }
241 
242  unsigned char width_field;
243  if (sign) {
244  width_field = 0x80u >> num_bytes;
245  } else {
246  width_field = 0x80u | ~(0xFFu >> num_bytes);
247  }
248  auto uobj = static_cast<std::uint32_t>(obj);
249  unsigned char value_mask = (0xFFu >> (num_bytes + 1));
250  unsigned char high_byte = (num_bytes == 5 ? (sign ? 0xFF : 0) : (uobj >> ((num_bytes - 1) * 8)));
251  stream.write(width_field | (high_byte & value_mask));
252  for (int i = num_bytes - 2; i >= 0; --i) { stream.write(static_cast<char>((uobj >> i * 8) & 0xFFu)); }
253 }

287 template <typename UInt, typename T>
288 UInt float_to_key(T value) {
289  static_assert(sizeof(T) == sizeof(UInt), "Expected unsigned int of the same size");
290  UInt result;
291  std::memcpy(&result, &value, sizeof(T));
292  UInt signbit = (static_cast<UInt>(1) << (std::numeric_limits<UInt>::digits - 1));
293  UInt mask = 0;
294  if (result == signbit)
295  result = 0;
296  if (result & signbit)
297  mask = ~mask;
298  return result ^ (mask | signbit);
299 }

301 template <typename T, typename S>
302 void to_key(const T& obj, datastream<S>& stream) {
303  if constexpr (std::is_floating_point_v<T>) {
304  if constexpr (sizeof(T) == 4) {
305  to_key(float_to_key<uint32_t>(obj), stream);
306  } else {
307  static_assert(sizeof(T) == 8, "Unknown floating point type");
308  to_key(float_to_key<uint64_t>(obj), stream);
309  }
310  } else if constexpr (std::is_integral_v<T>) {
311  auto v = static_cast<std::make_unsigned_t<T>>(obj);
312  v -= static_cast<std::make_unsigned_t<T>>(std::numeric_limits<T>::min());
313  std::reverse(reinterpret_cast<char*>(&v), reinterpret_cast<char*>(&v + 1));
314  stream.write(&v, sizeof(v));
315  } else if constexpr (std::is_enum_v<T>) {
316  static_assert(!std::is_convertible_v<T, std::underlying_type_t<T>>, "Serializing unscoped enum");
317  to_key(static_cast<std::underlying_type_t<T>>(obj), stream);
318  } else {
319  using ts_meta = bluegrass::meta::meta_object<T>;
320  ts_meta::for_each_field(obj, [&](const auto& member) {
321  to_key(member, stream);
322  });
