144 constexpr name suffix()const {
145 uint32_t remaining_bits_after_last_actual_dot = 0;
146 uint32_t tmp = 0;
147 for( int32_t remaining_bits = 59; remaining_bits >= 4; remaining_bits -= 5 ) { // Note: remaining_bits must remain signed integer
148 // Get characters one-by-one in name in order from left to right (not including the 13th character)
149 auto c = (value >> remaining_bits) & 0x1Full;
150 if( !c ) { // if this character is a dot
151 tmp = static_cast<uint32_t>(remaining_bits);
152 } else { // if this character is not a dot
153 remaining_bits_after_last_actual_dot = tmp;
154 }
155 }
156
157 uint64_t thirteenth_character = value & 0x0Full;
158 if( thirteenth_character ) { // if 13th character is not a dot
159 remaining_bits_after_last_actual_dot = tmp;
160 }
161
162 if( remaining_bits_after_last_actual_dot == 0 ) // there is no actual dot in the %name other than potentially leading dots
163 return name{value};
164
165 // At this point remaining_bits_after_last_actual_dot has to be within the range of 4 to 59 (and restricted to increments of 5).
166
167 // Mask for remaining bits corresponding to characters after last actual dot, except for 4 least significant bits (corresponds to 13th character).
168 uint64_t mask = (1ull << remaining_bits_after_last_actual_dot) - 16;
169 uint32_t shift = 64 - remaining_bits_after_last_actual_dot;
170
171 return name{ ((value & mask) << shift) + (thirteenth_character << (shift-1)) };
