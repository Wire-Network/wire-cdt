365template<int I, typename Stream, typename... Ts>
366void deserialize(datastream<Stream>& ds, std::variant<Ts...>& var, int i) {
367 if constexpr (I < std::variant_size_v<std::variant<Ts...>>) {
368 if (i == I) {
369 std::variant_alternative_t<I, std::variant<Ts...>> tmp;
370 ds >> tmp;
371 var.template emplace<I>(std::move(tmp));
372 } else {
373 deserialize<I+1>(ds,var,i);
374 }
375 } else {
376 sysio::check(false, "invalid variant index");
377 }
